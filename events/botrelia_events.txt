# Guns to Botrelia
country_event = {
	id = botrelia.12
	title = botrelia.12.t
	desc = botrelia.12.d
	picture = GFX_report_event_merchant_ship_01
	
	is_triggered_only = yes

	
	option = { # Accept Deal
		name = botrelia.12.a
		ai_chance = { factor = 90 }
		add_equipment_to_stockpile = {
			type = infantry_equipment
			amount = 10000
		}
		random_owned_controlled_state = {
			limit = {
				free_building_slots = {
					building = arms_factory
					size > 0
					include_locked = yes
				}
			}
			add_extra_state_shared_building_slots = 1
			add_building_construction = {
				type = arms_factory
				level = 1
				instant_build = yes
			}
		}
		diplomatic_relation = {
			country = POL
			relation = non_aggression_pact
			active = yes
		}
		POL = { country_event = { id = pithia.7 days = 1 } }
	}
	option = { # Reject Deal
		name = botrelia.12.b
		ai_chance = { factor = 10 }
		POL = { country_event = { id = pithia.8 days = 1 } }
	}
}
